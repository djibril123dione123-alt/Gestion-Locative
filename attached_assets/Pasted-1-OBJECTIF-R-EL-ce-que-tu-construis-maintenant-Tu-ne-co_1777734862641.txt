🥇 STEP 1 — financial_snapshots (OBLIGATOIRE)
CREATE TABLE financial_snapshots (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  agency_id uuid NOT NULL,
  period date NOT NULL,

  total_paiements numeric DEFAULT 0,
  total_ledger numeric DEFAULT 0,

  diff numeric GENERATED ALWAYS AS (
    total_paiements - total_ledger
  ) STORED,

  status text DEFAULT 'ok',

  created_at timestamp DEFAULT now()
);
🥇 STEP 2 — event emitter central

src/lib/eventBus.ts

export const emitEvent = async (event) => {
  await supabase.from('event_log').insert(event)

  if (window.posthog) {
    posthog.capture(event.type, event.payload)
  }
}
🥇 STEP 3 — state machine business rules

Ajouter :

src/domain/stateMachine.ts

Exemple :

export const validatePaiementTransition = (from, to) => {
  const allowed = {
    pending: ["paid", "cancelled"],
    paid: ["refunded"],
  }

  if (!allowed[from]?.includes(to)) {
    throw new Error("Invalid state transition")
  }
}
